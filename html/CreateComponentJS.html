import React from "react";

function Tweet() {
  return React.createElement(
    "article",
    {
      className: "tweet"
    },
    React.createElement(
      "header",
      {
        className: "tweet__header"
      },
      React.createElement("img", {
        className: "tweet__header-avatar",
        src: "https://api.adorable.io/avatars/64/react@adorable.png",
        alt: "Avatar"
      }),
      React.createElement(
        "h2",
        {
          className: "tweet__header-name"
        },
        "React"
      )
    ),
    React.createElement(
      "main",
      {
        className: "tweet__content"
      },
      React.createElement(
        "p",
        null,
        "A JavaScript library for building user interfaces"
      )
    ),
    React.createElement(
      "footer",
      {
        className: "tweet__footer"
      },
      "May 29th, 2013"
    )
  );
}

This is why we have to import React from 'react' even when we don't see a reference to React anywhere in the file. When the JSX is converted to JavaScript the dependency on React is revealed.
